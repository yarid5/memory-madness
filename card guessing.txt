<script>
const emojis = ['‚öΩ','üö¨','üí£','üë®üèø‚Äçü¶≤','üçÜ','ü•ä']
let cardsArray = [...emojis, ...emojis];
let flippedCards = [];
let matchedCount = 0;

function shuffle(array) {
  return array.sort(() => Math.random() - 0.5);
}

function createBoard() {
  const board = document.getElementById('game-board');
  board.innerHTML = '';
  shuffle(cardsArray).forEach((emoji, index) => {
    const card = document.createElement('div');
    card.classList.add('card');
    card.dataset.emoji = emoji;
    card.dataset.index = index;
    card.textContent = '‚ùì';
    card.addEventListener('click', flipCard);
    board.appendChild(card);
  });
}

function flipCard() {
  if (flippedCards.length < 2 && this.textContent === '‚ùì') {
    this.textContent = this.dataset.emoji;
    flippedCards.push(this);

    if (flippedCards.length === 2) {
      setTimeout(checkMatch, 500);
    }
  }
}

function checkMatch() {
  const [card1, card2] = flippedCards;
  if (card1.dataset.emoji === card2.dataset.emoji) {
    matchedCount += 2;
    if (matchedCount === cardsArray.length) {
      celebrateWin();
    }
  } else {
    card1.textContent = '‚ùì';
    card2.textContent = '‚ùì';
  }
  flippedCards = [];
}

function celebrateWin() {
  document.body.style.backgroundColor = '#FFD700';
  alert('üéâ Congratulations! You matched all cards! üéâ');
}

function resetGame() {
  matchedCount = 0;
  flippedCards = [];
  createBoard();
  document.body.style.backgroundColor = '#2e8b57';
}

createBoard();
